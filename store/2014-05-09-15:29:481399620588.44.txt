<script type="text/javascript">
var appID=1412796;
var reportTimer;
$().ready(function(){
    var oUl = document.getElementById('screenCont');
    var aLi = oUl.getElementsByTagName('li');
    var oBtn_Prev = document.getElementById('pBtnPrev');
    var oBtn_Next = document.getElementById('pBtnNext');
    var total=aLi.length;
    var iNow = 0;
    var i = '';

    setBtn();
    oBtn_Prev.onclick = function(){
        if( iNow &gt; 0 ){
            iNow--;
            stormStartMove( oUl, { left: -((aLi[0].offsetWidth+28) * iNow ) }, STORM_MOVE_TYPE.BUFFER );
            setBtn();
        }
    };
    oBtn_Next.onclick = function(){
        if( iNow &lt; total - 2 ){   // 动画的点击次数
            iNow++;
            stormStartMove( oUl, { left: -((aLi[0].offsetWidth+28) * iNow ) }, STORM_MOVE_TYPE.BUFFER );
            setBtn();
        }
    };
    function setBtn()
    {
        oBtn_Prev.className = (total==0 || iNow == 0) ? 'avcPrevNa' : 'avcPrev';
        oBtn_Next.className = (total&lt;=2 || iNow == ( total - 2 )) ?  'avcNextNa' : 'avcNext';
    }
    
    $("#description_more").click(function(){
        $(this).hide();
        $("#description_p").css("height",'auto');
    });
    $(".switch_basic").click(function(){
        if($(this).hasClass("on"))return;

        $(this).parent().find('li').removeClass("on");
        $(this).addClass("on");

        var i=$(this).index();
        if(i==0){$("#basic_info").show();}else{$("#basic_info").hide();}
        $("#cmt_page").html($("#cmt_page").html().replace(/(&amp;|\?)sc=\d(&amp;|")/g,"$1sc="+i+"$2"));
    });
    $("#rec_more").bind('click',recMore);
    $("#sel_score a").bind("click",function(){doScore($(this))});
    initAppReport();
});

function recMore()
{
    var total=$("#rec_list li").length;
    var curIdx=$("#rec_list li:visible:last").index();
    var newSIdx=(curIdx+1)&gt;=total?0:curIdx;
    $("#rec_list li").css('display','none');
    if(newSIdx==0)
        $("#rec_list li:lt(6)").css("display","block");
    else
        $("#rec_list li:gt("+newSIdx+"):lt(6)").css("display","block");
}
function doScore(e)
{
    var score=e.index()+1;
    $.ajax({
        type:'post',
        url:'/ajax_score.php',
        data:'appID='+appID+'&amp;score='+score+'&amp;f=p_wenjiao',
        dataType:'json',
        success:function(result){
            var msg=result.msg;
            if(result.errno==4 || result.errno==5)
            {
                score=result.score;
                $("#rating_box").html('&lt;span class="rateValueL"&gt;&lt;span class="rateStarL rateStarL-'+score+'"&gt;已评分&lt;/span&gt;&lt;/span&gt;');
            }
            else if(result.errno==1)
                $("#login_btn").click();
        }
    });
}
function initAppReport()
{
    $("#report_close,#report_cancle").click(function(){closeReportModal();});
    $("#report_do").bind("click",doReport);
    $("#report_cont input[type=checkbox]").bind("click",function(){
        $("#reporting_warning").hide();
    });
    $("#btn_report").bind("click",function(){$("#report_window").modal({opacity:60,overlayCss: {backgroundColor:"#000"}});});
}
function doReport()
{
    var types='';
    $("#report_cont input[type=checkbox]").each(function(){
            if($(this).attr("checked")=="checked")
                types+=$(this).val()+",";
        }
    );
    types=types.replace(/,$/,'',types);
    if(types=="")
    {
        $("#reporting_warning").show();
        return;
    }

    $("#report_cont").hide();
    $("#report_msg").html("正在提交...");
    $("#report_timer span:eq(0)").html(5);
    $("#report_status").show("");
    $.ajax({
        type:'post',
        url:'/ajax_report.php',
        data:'appID='+appID+'&content;='+types+'&f;=p_wenjiao',
        dataType:'json',
        success:function(result){
                $("#report_msg").html(result.msg);
            },
            error:function(){
                $("#report_msg").html("亲，出问题了，稍后再试一下吧...");
            },
            complete:function(){
                reportTimer=setInterval(function(){
                    var s=parseInt($("#report_timer span:eq(0)").html());
                    var sN=s-1;
                    if(sN&lt;=0)
                        closeReportModal();
                    $("#report_timer span:eq(0)").html(sN);
                },1000);
            }
    });
}
function closeReportModal()
{
    if(reportTimer)
        clearInterval(reportTimer);
    $.modal.close();
}
