      _gz = false;

  var num = 60;

  var str = '请填写手机号码';
  var clock = function(){
    num--;
    if (num &gt; 0) {
      $btn.text(num + '秒后可重发');

      clearInterval(numTime);
      $btn.text('发 送').removeClass('undo');
      _gz = true;
    }
  };
  
  if ($input.val()=='' || $input.val()==str) {
    $btn.addClass('undo');
  }else{
    $btn.removeClass('undo');
    _gz = true;

  if ($input.val()==str) {
    $input.css('color','#ccc');
  }else{
    $input.css('color','#000');
  }

  $input.focus(function(){
    if ($(this).val()==str) {
      $(this).css('color','#000');
      $(this).val('');
    }
  })

  $input.focusout(function(){
    if ($(this).val()=='') {
      $(this).css('color','#ccc');
      $(this).val(str);
    }
  })

    $(this).val($(this).val().replace(/\D/g,''));
