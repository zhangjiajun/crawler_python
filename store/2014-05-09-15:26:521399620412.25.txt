function subCom(f){
	$(f).ajaxSubmit({
		type: 'POST',　　
   		url: '/org/comment',
		success: function(response){
			if(response.status==1){
				$('textarea[name="com"]').val('');
				$.getJSON('/org/WeiboNum?id='+response.mid+'&amp;callback=?', function (res) {
					$('#'+response.f+'_n_'+response.mid).html( res.com_n );
				})
			}else{
				$('textarea[name="com"]').css("background","#FF8080");
				$('textarea[name="com"]').animate({"opacity":"50"},{"complete":function(){
					$('textarea[name="com"]').css("background","#ffffff");

				}},1000);
			}
		},
   		dataType: 'json'
	});
	return;
}
