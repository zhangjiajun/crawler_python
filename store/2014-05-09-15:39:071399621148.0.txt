function getSite(site) {
	var arr;
	if(site!=""&amp;&amp;site.indexOf(',')&gt;-1){//老数据逗号分隔
		arr = site.split(',');
	}else{//新数据顿号分隔
		arr = site.split('、');
	}
    var c = arr.length;
    if(c &lt; 3) { c = 3; }
    if( c &gt; 5) {c = 5; }
    
    var htm = [];
    for(var i =0; i &lt; c; i ++) {
        htm.push('&lt;input type=text idref="site" class="text5" value="' + (i &lt; arr.length ? arr[i] : '') + '" style="width:50px" maxlength="10"  /&gt;');
		if(i != c-1) htm.push(' - ');
    }
	
	// 添加更多站点
	if(c != 5) {
		htm.push('&lt;a href="javascript:void(addSite());" id="btnAddSite"&gt;添加更多站点&lt;/a&gt;');
	}
	
	var td = $('#siteq td:eq(1)');
	td.prepend(htm.join(''));
}

	}
}
