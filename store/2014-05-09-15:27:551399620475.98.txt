        function url_set_search(key,value,clear){
            var search = window.location.search;
            var pathname = window.location.pathname;
            var keyValue = new Array();
            if(search &amp;&amp; search.charAt(0) == '?'){
                search = search.substr(1);
            }
            var flag = false;
            if(search &amp;&amp; !clear){
                var arr = search.split('&amp;');
                for(var i =0;i&lt;arr.length;i++){
                    var tmp = arr[i].split('=');
                    if(tmp[0]==key){
                        if(value){
                            keyValue.push(key+"="+encodeURIComponent(value));
                        }
                        flag = true;
                    }
                    else{
                        keyValue.push(arr[i]);
                    }
                }
            }
            if(flag == false){
                keyValue.push(key+"="+encodeURIComponent(value));
            }
            search = keyValue.join("&amp;");
            search = search?"?"+search:"";
            window.open(pathname+search,'_self');

            if(event){
                if (event.stopPropagation)
                event.stopPropagation();
                if (event.preventDefault)
                event.preventDefault();
                event.cancelBubble = true;
                event.returnValue = false;
            }
            return false;
        }
        function set_page_num(number){
            url_set_search('p',number,false);
        }
