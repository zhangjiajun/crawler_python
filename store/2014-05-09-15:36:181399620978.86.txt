    // 验证标题
    function checkPhoneTitle(){
        var result;
        var err_empty = '';
        var pattern = '';
        var err_msg = '';
        if (getTypeIndex() == 0) {
            err_empty = '号码不能为空';
            pattern = '^(13|14|15|18)\\d{9}$';
            err_msg = '请正确填写手机号码';
            var _lx = $.c.common.getSelectedText('ObjectType'); // 类型
            if(_lx == '小灵通' || _lx == '铁通'){
                pattern = '^[1-9](\\d{6}|(\\d{7})|(\\d{10}))$';
                err_msg = '请正确填写' + _lx + '号码';
            }
        } else {
            err_empty = '标题不能为空';
            pattern = '^[\\s\\S]*$#^[\\s\\S]{6,}$#^(.){6,40}$';
            err_msg = '请输入标题#标题不足6个字#6-40个字，不能填写电话和特殊符号！';
        }
        // 空值验证
        var val = $('#Title').val();
        if(val.length == 0){
            return err_empty;
        }
        // 格式验证
        var arrPattern = pattern.split('#');
        var arrError = err_msg.split('#');
        for(var i = 0; i &lt; arrPattern.length; i ++){
            if(!val.match(new RegExp(arrPattern[i]))) {
                return arrError[i];
            }
        }
        return true;
    }
