// 获取区间索引
function getQuJianIndex(inputVal, qjParaName){
	var index = -1;
	if(inputVal == null || inputVal.length == 0) return index; // 空值
	if(isNaN(inputVal)) return index; // 非数字
	
	var inputNum = parseInt(inputVal, 10);
	var obj = $('#' + qjParaName);
	if(obj.length &gt; 0){
		var d = $.cmcs.getParaItem(qjParaName);
		if(d != null){
			$(d.value).each(function(i, e) {
				var arr = e.v.split('_');
				if (arr.length == 2) {
					var a = parseInt(arr[0]);
					var b = parseInt(arr[1]);
					if (inputNum &gt;= a &amp;&amp; inputNum &lt;= b) {
						index = i;
						return false;
					}
				}
			});
		}
	}
	return index;
}

function setQuJianIndex(inputPara, qjPara){
	var c1 = $('#' + inputPara);
	var c2 = $('#' + qjPara);
	if(c1.length &gt; 0 &amp;&amp; c2.length &gt; 0){
		c2.parent().parent().hide(); // 隐藏区间
		c1.bind('blur', function(){
			var index = getQuJianIndex(c1.val(), qjPara);
			c2.get(0).selectedIndex = index + 1;
		});
	}
}

// 调用示例：
$().ready(function(){
	setQuJianIndex('MinPrice', 'MinPriceqj');
});
