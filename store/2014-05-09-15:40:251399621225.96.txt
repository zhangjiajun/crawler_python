function show(ls, attr, index, jsonLength)  //  记录flashCookie
{
    var flash = get_flash("myFlash");
    var ls_obj = null;

    if (index == 0) {
        ls_obj = ls.browse;
    }
    else if (index == 1) {
        ls_obj = ls.place;
    } else {
        return;
    }
    if (ls_obj.length == 0) {
        ls_obj.splice(0, 0, attr);
        flash.writeCookie($.toJSONString(ls))
    } else {
        if (ls_obj.length &gt;= jsonLength) {
            ls_obj.length = jsonLength;
        }
        for (var i = 0; i &lt; ls_obj.length; i++) {
            if (ls_obj[i].title_A == attr.title_A) {
                attr = ls_obj.splice(i, 1)[0];
            }
        }
        ls_obj.splice(0, 0, attr);
        flash.writeCookie($.toJSONString(ls))
    }
}
function getEventTarget(ev) {
    var oEvent = ev || event;

    }
    else {

    }
