function chkDate(){ //检查时间格式
	if(this.value !== ''){
		var err = false;
		var data = this.value.match(/^(\d{4})\-0?(\d{1,2})\-0?(\d{1,2})$/i);
		if(data){
            current_date = new Date();
			if(data[1] &lt; 1998 || data[1] &gt; current_date.getFullYear()){err = true};
			if(data[2] == 0 || data[2] &gt; 12){err = true};
			if(data[3] == 0 || data[3] &gt; 31){err = true};
			if(data[2] == 2 &amp;&amp; data[3] &gt; 28){
				err = true;
				if(data[1]%4==0 &amp;&amp; data[3] == 29){err = false};
			};
			if((data[2] == 4 || data[2] == 6 || data[2] == 9 || data[2] == 11) &amp;&amp; data[3] &gt; 30){err = true};
		}else{
			err = true;
		}
	}else{
		err = false;
	}
	
	if(err){
		if($("#" + this.id + "_err").length == 0){
			$("#" + this.id).after('&lt;span id="' + this.id + '_err" style="font-size:12px;color:#f00;line-height:20px;display:block;"&gt;您选择的日期不合理&lt;/span&gt;');
		};
		$("#" + this.id + "_err").css('display','block');
	}else{
		$("#" + this.id + "_err").css('display','none');
	};

			var tempThis = this;
			this.addEvent(this.pageObj,'touchstart',function(e){tempThis._touchstart(e)});
			this.addEvent(this.pageObj,'touchmove',function(e){tempThis._touchmove(e)});
			this.addEvent(this.pageObj,'touchend',function(e){tempThis._touchend(e)});
	   },
_touchstart : function(e){
		      this.x = e.touches[0].pageX;
		      this.scrollX = window.pageXOffset;
		      this.scrollY = window.pageYOffset; //用于判断页面是否滚动
	      },
_touchmove : function(e){
		     if(e.touches.length &gt; 1){ //多点触摸
			     this.status = 'ok';
			     return;
		     };
		     this.lastX = e.touches[0].pageX;
		     var cX = this.x - this.lastX;

		     if(cX&lt;0){//第一页禁止向左
			     if(!this.prePage){
				     return;
			     };
		     };
		     if(this.status == 'ok'){
			     if(this.scrollY == window.pageYOffset &amp;&amp; this.scrollX == window.pageXOffset &amp;&amp; Math.abs(cX)&gt;50){ //横向触摸
				     if(cX&gt;0){//最后一页禁止向右
					     if(!this.nextPage){
						     return;
					     };
				     };
				     if(!this.parentObj){
					     this.parentObj = document.createElement('div');
					     this.parentObj.style.width = this.pageObj.offsetWidth + 'px';
					     this.parentObj.style.overflow = 'hidden';
						 this.pageObj.style.width = this.pageObj.offsetWidth + 'px';
					     this.pageObj.style.overflow = 'hidden';
					     this.pageObj.parentNode.insertBefore(this.parentObj,this.pageObj);
					     this.parentObj.appendChild(this.pageObj);
				     }
				     this.status = 'touch';
			     }else{
				     return;
			     };
		     };

		     this.pageObj.style.marginLeft = - cX + 'px';
		     e.preventDefault();
	     },
_touchend : function(e){
		    if(this.status != 'touch'){return};
		    this.status = 'ok';
		    var cX = this.x - this.lastX;
		    var tempThis = this;
		    if(this.alertStatus == 'open'){this.hideAlert()};
		    this.actPX(this.pageObj.style,'marginLeft',this.pageObj.offsetLeft,cX&gt;0?-(this.parentObj.offsetWidth+2):this.parentObj.offsetWidth+2,200,function(){
				    if(cX&lt;0){
				    if(tempThis.prePage){window.location.href = tempThis.prePage}else{tempThis.pageObj.style.marginLeft = 0};
				    }else{
				    if(tempThis.nextPage){window.location.href = tempThis.nextPage}else{tempThis.pageObj.style.marginLeft = 0};
				    };
				    });
	    },
actPX : function(obj,key,start,end,speed,endFn,u){
		if(typeof(u) == 'undefined'){u = 'px'};
		clearTimeout(obj['_extend_actPX' + key.replace(/\-\.\=/,'_') + '_timeOut']);
		if(start &gt; end){
			speed = - Math.abs(speed);
		}else{
			speed = Math.abs(speed);
		};
		var now = start;
		var length = end - start;
		obj['_extend_actPX' + key.replace(/\-\.\=/,'_') + '_timeOut'] = setTimeout(function(){
				now += speed;
				//debugger;
				var space = end - now;
				if(start &lt; end){
				if(space &lt; length/3){
				speed = Math.ceil(space/3);
				};
				if(space &lt;= 0){ //end
				obj[key] = end + u;
				if(endFn){endFn()};
				return;
				};
				}else{
				if(space &gt; length/3){
				speed = Math.floor(space/3);
				};
				if(space &gt;= 0){ //end
				obj[key] = end + u;
				if(endFn){endFn()};
				return;
				};
				};

				obj[key] = now + u;
				obj['_extend_actPX' + key.replace(/\-\.\=/,'_') + '_timeOut'] = setTimeout(arguments.callee,20);

		},20);
	},
