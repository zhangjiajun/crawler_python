// 接收图片控件的值
function getPicValueForPost() {
    var pic = [];
    var tag = [];
    var desc = [];
    var data = UpImageShowBar.getImgs();
    if(data != null &amp;&amp; data.length &gt; 0) {
        $(data).each(function(i, n){
            // 标签
            var _tag = n.label;
            if(_tag == null) _tag = '';
            tag.push(_tag);

            // 图片路径
            var _url = n.url;
            _url = _url.replace('http://pic.58.com', '').replace('/tiny/', '/big/');
            pic.push(_url);
            desc.push(n.detail);
            // 描述
        });
    }
    return { "pic" : pic, "tag" : tag, "desc" : desc };
}

// 图片元素赋值
function setPicValueForPost() {
    var data = getPicValueForPost();
    $.bll.xiaobao.create('pic', data.pic.join('|'));
    $.bll.xiaobao.create('pictag', data.tag.join('|'));
    $(data.desc).each(function(i,n){
        var id = 'picdesc' + (++i);
        $.bll.xiaobao.create(id, n);
    });
}

	function cacuxingjiabi(){
			var xingjiabi = 0;
			if(jiagecankao==1){//有价格参考
				var price = $("#MinPrice").val();
				var oldlevel = $("#oldlevel").val();
				var score = $("#jiagexiuzheng").val();
				xingjiabi = execacu(price,oldlevel,score,mincankaoprice,maxcankaoprice);
				$("#xingjiabifen").val(xingjiabi);
			}else if(jiagecankao==-1){//判断是否有价格参考
				var url = buildcankaourl();
				if(url!=null&amp;&url;!=""){
    				var price = $("#MinPrice").val();
    				var oldlevel = $("#oldlevel").val();
    				var score = $("#jiagexiuzheng").val();
    				$("#xingjiabifen").val(url+"|"+price+"|"+oldlevel+"|"+score);
				}else{
					$("#xingjiabifen").val("0");
				}
			}else //价格参考返回值是空
				$("#xingjiabifen").val(0);
	}	
	function execacu(price,oldlevel,score,mincankao,maxcankao){
		var scorea = 0;//物品报价为p，参考价格区区间a~b，参考价格区间中间值c=（a+b）/2
		var scoreb = 0;
		var scorec = 0;
		var scored = 0;
		
		//物品价格相对因素（分值A	）
		if(price!=0&amp;&mincankao;!=0&amp;&maxcankao;!=0){
		   var rate = $("#MinPrice").val()/((mincankao+maxcankao)/2);
		   rate = rate.toFixed(2);
		   if(rate&gt;=0.85&amp;&rate;&lt;=1.1)
		   	scorea = ((1-rate).toFixed(2))*200;
		   else if(rate&gt;=0&amp;&rate;&lt;0.85)
		   	scorea = 0;
		   else if(rate&gt;1.1)
		   	scorea = -30;
		}
		
		//价格绝对因素（分值B	）
		if(500&lt;=price&amp;&price;&lt;=1600)
			scoreb = 5;
		else if((300&lt;=price&amp;&price;&lt;500)||(1600&lt;price&gt;&lt;=2000))
			scoreb = 0;
		else if(price&lt;300||price&gt;2000)
			scoreb = -5;
			
		//物品参数新旧程度（分值C	）
		if(oldlevel == 2){//95成新时，分值C= 15
			scorec = 15;
		}else if(oldlevel == 3||oldlevel==7){//9成新，全新时，分值C=10
			scorec = 10;
		}else if(oldlevel == 4){//8成新时，分值C=0
			scorec = 0;
		}else if(oldlevel == 5){//7成及以下时分值C= -10
			scorec = -10;
		}else{
			scorec = 0;
		}
		
		//物品功能状况分（分值D	） 
		if(score==null||score==''||score-0&gt;=0)
			scored = 0;
		else
			scored = -10;
		return scorea+scoreb+scorec+scored;
	}
