//获取匹配车系、车型等下拉框的标题字符串
function getTitleStr(){
	var v1 = '', v2 = '', v3 = '',str = '',value3='';
    var c1 = $('#brand');
    var c2 = $('#chexi');
    var c3 = $('#carchexing');
    
    // 品牌
    if(c1.length &gt; 0){
        if(c1.get(0).selectedIndex &gt; 0){
            v1 = c1.get(0).options[c1.get(0).selectedIndex].text;
        }
    }
    
    // 车系
    if(c2.length &gt; 0){
        if(c2.get(0).selectedIndex &gt; 0){
            v2 = ' ' + c2.get(0).options[c2.get(0).selectedIndex].text;
        }
    }
    
    // 车型
    if(c3.length &gt; 0){
        if(c3.get(0).selectedIndex &gt; 0){
            v3 = ' ' + c3.get(0).options[c3.get(0).selectedIndex].text;
            value3 = c3.get(0).options[c3.get(0).selectedIndex].value;
        }
    }
    var Vchexibieming = '';
    //车系别名
	var chexibiemingObj = document.getElementById("chexibieming");
    if(chexibiemingObj != null){
    	Vchexibieming = chexibiemingObj.options[chexibiemingObj.selectedIndex &gt; -1 ? chexibiemingObj.selectedIndex : 0].text;
    	var cxbiemingValue = chexibiemingObj.options[chexibiemingObj.selectedIndex &gt; -1 ? chexibiemingObj.selectedIndex : 0].value;
    	if(cxbiemingValue != ""){
        	if(value3 != ""){
        		str = Vchexibieming + v3;
        	}else{
        		str = Vchexibieming;
        	}
    	}
    }else if(value3 != ""){
    	str = v2 + v3;
    }else{
    	str = v2;
    }
    //调用标题优化新需求
    var strObj=setTitleZdy(v1,v2,v3,value3);
    if(strObj.b){str=strObj.str}
    return str;
}

function setTitleZdy(v1,v2,v3,value3){
    var bieming="";
    var vstr="";
    var b=false;
    //处理车系别名

        bieming= $("#chexibieming")[0].options[$("#chexibieming")[0].selectedIndex &gt; -1 ? $("#chexibieming")[0].selectedIndex : 0].text;
    }else{
        bieming=v1+v2;
    }
    if(value3 == ""&amp;&amp;$('#carchexing').length&gt;0){//自定义车型
        vstr=bieming+" "+$("#cheXing").val()+" "+diaoyongSpnf();
       b=true;
    }
    if($("#chexi").length&gt;0&amp;&amp;$('#carchexing').length&lt;=0){//有车系没有车型的时候
        vstr=bieming+" "+diaoyongSpnf();
        b=true;
    }
    return{
        b:b,
        str:vstr
    }
