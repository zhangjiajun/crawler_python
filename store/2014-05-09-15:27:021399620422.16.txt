function initPhoneSelect(manu)
{
	var dltObj = document.getElementById('phoneselect');
	dltObj.innerHTML=''; 
	$.ajax({
         	url: '/interface/3g/3gsoftphone.php?m='+encodeURIComponent(manu),
         	type: 'get',
         	dataType: 'json',
         	timeout: 61000,
         	error: function(){
         		alert('Please check web status');
         	},
         	success: function(data){
         		//alert(data);
         		
         		for ( var i in data )
				{
				    var opt=document.createElement("OPTION"); 
					dltObj.options.add(opt); 
					opt.value = data[i].d_id; 
					opt.text = data[i].name; 
				}
         	}
         });
}
