    function refrashPriceQj() {
        var price = $('#esqgjiage').val();
        var qj = $('#esqgjiageqj');
        if (qj.length == 0) return;
        if (price == '' || isNaN(price)) {
            qj.get(0).selectedIndex = 0;
        } else {
            $('#esqgjiageqj option').each(function(i, n) {
                var str = n.value;
                if (!!str) {
                    var arr = str.split('_');
                    if (arr.length == 2) {
                        var p1 = parseInt(arr[0]);
                        var p2 = parseInt(arr[1]);
                        var p = parseInt(price);
                        if (p &gt;= p1 &amp;&amp; p &lt;= p2) {
                            $(qj).val(str);
                            return false;
                        }
                    }
                }
            });
        }
    }
