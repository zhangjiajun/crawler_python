function getPicValueForPost() {
    var pic = [];
    var tag = [];
    var desc = [];
    var data = UpImageShowBar.getImgs();
    if(data != null &amp;&amp; data.length &gt; 0) {
        $(data).each(function(i, n){
            // 标签
            var _tag = n.label;
            if(_tag == null) _tag = '';
            tag.push(_tag);

            // 图片路径
            var _url = n.url;
            _url = _url.replace('http://pic.58.com', '').replace('/tiny/', '/big/');
            pic.push(_url);
            desc.push(n.detail);
            // 描述
        });
    }
    return { "pic" : pic, "tag" : tag, "desc" : desc };
}

// 图片元素赋值
function setPicValueForPost() {
    var data = getPicValueForPost();
    $.bll.xiaobao.create('pic', data.pic.join('|'));
    $.bll.xiaobao.create('pictag', data.tag.join('|'));
    $(data.desc).each(function(i,n){
        var id = 'picdesc' + (++i);
        $.bll.xiaobao.create(id, n);
    });
}

        $('#MinPriceqj').hide();
        $('#MinPriceqj').parent().parent().hide();
    });
    function refrashPriceQj() {
        var price = $('#MinPrice').val();
        var qj = $('#MinPriceqj');
        if (qj.length == 0) return;
        if (price == '' || isNaN(price)) {
            qj.get(0).selectedIndex = 0;
        } else {
            $('#MinPriceqj option').each(function(i, n) {
                var str = n.value;
                if (!!str) {
                    var arr = str.split('_');
                    if (arr.length == 2) {
                        var p1 = parseInt(arr[0]);
                        var p2 = parseInt(arr[1]);
                        var p = parseInt(price);
                        if (p &gt;= p1 &amp;&amp; p &lt;= p2) {
                            $(qj).val(str);
                            return false;
                        }
                    }
                }
            });
        }
    }
