function searchResult(){
	var url = window.location.href;
	var url_arr = url.split("?");
	url = url_arr[0] + "?_action=t";

		url += "&amp;provid=" + $("#provid_select").val();
	}
	if ($("#cat_select").val() != ""){
		if ($("#cat_select").val() == "本科"){
			url += "&amp;benzhuan=1";
		}
		else if ($("#cat_select").val() == "专科"){
			url += "&amp;benzhuan=2";

		else if ($("#cat_select").val() == "独立"){
			url += "&amp;is_independ=1";
		}
		else if ($("#cat_select").val() == "教育部直属"){
			url += "&amp;is_zhishu=1";
		}
		else if ($("#cat_select").val() == "211工程"){
			url += "&amp;is211=是";
		}
		else if ($("#cat_select").val() == "985工程"){
			url += "&amp;is985=是";
		}
		else if ($("#cat_select").val() == "开设研究生院高校"){
			url += "&amp;is_yjs=1";
		}
		else if ($("#cat_select").val() == "民办高校"){
			url += "&amp;is_minban=1";
		}
	}
	if ($("#type_select").val() != ""){
		url += "&amp;type=" + $("#type_select").val();
	}
	if ($("#search_key").val() !="" &amp;&amp; $("#search_key").val() != "输入院校名称"){
		url += "&amp;name=" + $("#search_key").val();
	}
	window.location.href = url;
}
