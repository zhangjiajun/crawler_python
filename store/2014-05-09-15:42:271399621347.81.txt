				featureboxes: new Array,

				update: function(index, ticker) {
					if (!this.featureboxes[index]) {
						this.featureboxes[index] = {
							ticker: ticker,
							chart: false,
							size: false
						};
					}

					if (!this.ticker_data[ticker])
						this.ticker_data[ticker] = {};

					if (webDriver.jQuery("#ir_featurebox_"+index+"-quote").length)
						this.update_quote(index, ticker);

					if (webDriver.jQuery("#ir_featurebox_"+index+"-chart").length)
						this.update_chart(index, ticker);

					if (webDriver.jQuery("#ir_featurebox_"+index+"-data").length)
						this.update_data(index, ticker);

					this.update_labels(index, ticker);
				},

				update_labels: function(index, ticker) {
					if (!this.ticker_data[ticker]["labels"]) {
						webDriver.jQuery("#ir_featurebox_"+index+"-footer").html("");
						webDriver.jQuery.getJSON(
							"index.php",
							{
								ajax: "ir_featurebox",
								ticker: ticker,
								data: "labels"
							},
							function(data) {
								ir_featurebox.ticker_data[ticker]["labels"] = data;
								ir_featurebox.update_labels(index, ticker);
							}
						);
					} else {
						webDriver.jQuery("#ir_featurebox_"+index+"-footer").html(this.ticker_data[ticker]["labels"].footer);
					}
				}
			};
