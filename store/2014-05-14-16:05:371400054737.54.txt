	sPrint.BreakWord = function(str,intLen){
		var strTemp = "";
		if(str.length &gt; intLen){//考虑换行对其
			strTemp = "&lt;br  /&gt;";
		}
		while(str.length &gt; intLen){
			strTemp += str.substr(0,intLen)+"&lt;br  /&gt;";
			str = str.substr(intLen,str.length);
		}
		return strTemp += str;

	//处理正文html
	sPrint.HandlerBodyHtml = function(str){
		var strTmp = str;
		//滤掉 正文内部文字导航
		strTmp = strTmp.replace(/<!--\s+正文内部文字导航\s+:\s+begin -->[\s\S]*<!--\s+正文内部文字导航\s+:\s+end\s+-->/,"");
		//去除 artibody内打印容器以及以后的所有html内容
		strTmp = strTmp.replace(/&lt;span&gt;[\s\S]*/i, "");
		//过滤掉iask关键字的html标记
		strTmp = strTmp.replace(/&lt;span&gt;[\s\S]*?class=.?akey.?\s+title=[\s\S]*?&gt;([\s\S]*?)&lt;\/a&gt;&lt;\/span&gt;/gi, "$1");
		//绑定img onload
		strTmp = strTmp.replace(/&lt;img&gt;gi,"&lt;img onload=\"sPrint.ResizeImg(this);\"");
		return strTmp;
	}
