            // 标签

            if(_tag == null) _tag = '';
            tag.push(_tag);

            // 图片路径
            var _url = n.url;
            _url = _url.replace('http://pic.58.com', '').replace('/tiny/', '/big/');
            pic.push(_url);
            desc.push(n.detail);
            // 描述
        });
    }
    return { "pic" : pic, "tag" : tag, "desc" : desc };
}
